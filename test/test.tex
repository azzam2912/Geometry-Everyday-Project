Notasikan $\dangle$ sebagai directed angle.
Misalkan $O$ adalah pusat persegi $ABCD$. Jelas bahwa $DB \perp AC$. Lalu misalkan $T$ sebagai perpotongan $BF$ dan $AG$ Kita punya beberapa lemma berikut
\begin{lemma}
\label{ktomjan251}
    $A,B,G,O,F$ konsiklis.
    \begin{proof}
        $\dangle AFB = \dangle AOB = \dangle AGB = 90^\circ$ yang membuktikan $A,B,G,O,F$ konsiklis.
    \end{proof}
\end{lemma}
\begin{lemma}
\label{ktomjan252}
    $D,F,O,E$ konsiklis dan $E, O, G, C$ konsiklis.
    \begin{proof}
        Dengan bantuan Lemma~\ref{ktomjan251} dan fakta bahwa $AB \parallel CD$ didapat
        \begin{align*}
            \dangle DOF = \dangle BOF = \dangle BAF = \dangle DEF
        \end{align*}
        yang membuktikan $D,F,O,E$ konsiklis. Analog cara tersebut, $E, O, G, C$ juga konsiklis.
    \end{proof}
\end{lemma}
\begin{lemma}
\label{ktomjan253}
    $A,G,E,D$ konsiklis dan $F,B,E,C$ konsiklis.
    \begin{proof}
        $\dangle ADE = 90^\circ = \dangle AGE$ yang membuktikan $A,G,E,D$ konsiklis. $\dangle BFE = 90^\circ = \dangle BCE$ yang membuktikan $F,B,E,C$ konsiklis.
    \end{proof}
\end{lemma}
\begin{lemma}
\label{ktomjan254}
    $F,O,G,H$ konsiklis.
    \begin{proof}
        Dari Lemma~\ref{ktomjan251} dan Teorema Miquel pada $\triangle DHC$, didapatkan bahwa $O$ adalah Miquel point dari $\triangle DHC$ yang berakibat $F,O,G,H$ konsiklis.
    \end{proof}
\end{lemma}
Dari Lemma~\ref{ktomjan251} dan Lemma~\ref{ktomjan254}, didapat $F,A,H,B,G,O$ konsiklis.
