\begin{center}
    \begin{tikzpicture}[scale=2, dot/.style={circle, fill, inner sep=0.3pt, outer sep=0pt}]
    % MAIN Problem Picture 
    % Definisikan titik-titik A, B, C
    \tkzDefPoint(0,0){B}
    \tkzDefPoint(2,0){C}
    \tkzDefPoint(0.5,1.7){A}

    \tkzDrawSegments(B,C C,A A,B)
    \tkzLabelPoints[above](A)
    \tkzLabelPoints(B,C)
    
    \tkzDefTriangleCenter[circum](A,B,C)\tkzGetPoint{O}
    \tkzDrawCircle[thin](O,A)
 
    \tkzDefMidArc(O,C,B) \tkzGetPoint{S}
    \tkzLabelPoints[above](S)

    \tkzDefLine[perpendicular=through A](B,C) \tkzGetPoint{Aperp}
    \tkzInterLC(Aperp,A)(O,A) \tkzGetPoints{E0}{E}
    \tkzLabelPoints(E)
    \tkzDrawSegment(A,E)

    \tkzInterLL(Aperp,A)(B,S) \tkzGetPoint{D}
    \tkzLabelPoints(D)
    \tkzDrawPoint(D)

    \tkzDefLine[parallel=through D](B,C) \tkzGetPoint{Dpar}
    \tkzInterLL(Dpar,D)(B,E) \tkzGetPoint{L}
    \tkzLabelPoints[left](L)
    \tkzDrawSegment(L,D)

    % circumcircle BDL
    \tkzDefTriangleCenter[circum](B,D,L)\tkzGetPoint{O2}
    \tkzDrawCircle[red](O2,B)

    \tkzInterCC[common=B](O2,B)(O,A) \tkzGetFirstPoint{P}
    \tkzDrawPoint[red](P)
    \tkzLabelPoints[above](P)

    \tkzDefMidArc(O,B,C) \tkzGetPoint{F}
    \tkzLabelPoints(F)

    \tkzDrawSegments(E,L B,S A,F)

    % additional for solving
    \tkzInterLL(A,F)(B,S) \tkzGetPoint{Q}
    \tkzLabelPoints[right](Q)
    \tkzDrawPoints(Q)

    % segment PQ
    \tkzDrawSegments[thick, red](P,Q)

    % cicumcircle P,Q,B
    \tkzDefTriangleCenter[circum](P,Q,B)\tkzGetPoint{O3}
    \tkzDrawCircle[blue](O3,B)

    % intersection of (BPQ) with AF = intersection of PD with AH
    \tkzInterLL(A,F)(P,D) \tkzGetPoint{H}
    \tkzLabelPoints[right](H)
    \tkzDrawPoints(H)

    \tkzDrawSegments[dotted, thick, blue](H,P)

    % cicrcumcircle A,P,H
    \tkzDefTriangleCenter[circum](P,A,H)\tkzGetPoint{O4}
    \tkzDrawCircle[purple](O4,H)

    % intersection of (BPD) with AE = intersection of (BDL) and (APH)
    \tkzInterCC[common=P](O4,H)(O2,B) \tkzGetFirstPoint{G}
    \tkzLabelPoints(G)
    \tkzDrawPoints(G)

    \tkzDrawSegments[red](P,F P,A P,B S,F Q,O A,S P,S)

    \end{tikzpicture}
\end{center}
