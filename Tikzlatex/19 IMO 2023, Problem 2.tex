\begin{center}
    \begin{tikzpicture}[scale=2, dot/.style={circle, fill, inner sep=0.3pt, outer sep=0pt}]
    % MAIN Problem Picture 
    % Definisikan titik-titik A, B, C
    \tkzDefPoint(0,0){B}
    \tkzDefPoint(2,0){C}
    \tkzDefPoint(0.5,1.7){A}

    \tkzDrawSegments(B,C C,A A,B)
    \tkzLabelPoints[above left](A)
    \tkzLabelPoints(B,C)
    
    \tkzDefTriangleCenter[circum](A,B,C)\tkzGetPoint{O}
    \tkzDrawCircle[thin](O,A)
 
    \tkzDefMidArc(O,C,B) \tkzGetPoint{S}
    \tkzDrawPoint[fill=black, size=3](S)
    \tkzLabelPoints[above right](S)

    \tkzDefLine[perpendicular=through A](B,C) \tkzGetPoint{Aperp}
    \tkzInterLC(Aperp,A)(O,A) \tkzGetPoints{E0}{E}
    \tkzLabelPoints(E)
    \tkzDrawSegment(A,E)

    \tkzInterLL(Aperp,A)(B,S) \tkzGetPoint{D}
    \tkzLabelPoints(D)
    \tkzDrawPoint[fill=black, size=3](D)

    \tkzDefLine[parallel=through D](B,C) \tkzGetPoint{Dpar}
    \tkzInterLL(Dpar,D)(B,E) \tkzGetPoint{L}
    \tkzLabelPoints[left](L)
    \tkzDrawSegment(L,D)

    \tkzDefTriangleCenter[circum](B,D,L)\tkzGetPoint{O2}
    \tkzDrawCircle[thin, red](O2,B)

    \tkzInterCC[common=B](O2,B)(O,A) \tkzGetFirstPoint{P}
    \tkzDrawPoint[red, size=3](P)
    \tkzLabelPoints[below](P)

    \tkzDefMidArc(O,B,C) \tkzGetPoint{F}
    \tkzLabelPoints(F)

    \tkzDrawSegments(E,L B,S A,F)

    % additional for solving
    \tkzInterLL(A,F)(B,S) \tkzGetPoint{Q}
    \tkzLabelPoints[below right](Q)
    \tkzDrawPoints[green, size=3](Q)

    \tkzDrawSegments[red](P,Q)

    \tkzDefTriangleCenter[circum](P,Q,B)\tkzGetPoint{O3}
    \tkzDrawCircle[dashed, blue](O3,B)

    % intersection of (BPQ) with AF = intersection of PD with AH
    \tkzInterLL(A,F)(P,D) \tkzGetPoint{H}
    \tkzLabelPoints[right](H)
    \tkzDrawPoints[fill=black, size=3](H)

    % intersection of PD with (ABC)
    \tkzInterLC(P,D)(O,A) \tkzGetPoints{I0}{I}
    \tkzLabelPoints[right](I)
    \tkzDrawPoints[fill=black, size=3](I)

    % intersection of AE and IS
    \tkzInterLL(A,E)(I,S) \tkzGetPoint{X}
    \tkzLabelPoints[right](X)
    \tkzDrawPoints[fill=black, size=3](X)

    % intersection of AS and I
    \tkzInterLL(A,S)(I,P) \tkzGetPoint{Y}
    \tkzLabelPoints[above](Y)
    \tkzDrawPoints[fill=black, size=3](Y)

    \tkzDrawSegments(A,X S,Y Y,P A,I)
    \tkzDrawSegments[thick, blue](I,E I,P)
    \tkzDrawSegments[thick, green](X,I A,H)

    \end{tikzpicture}
\end{center}
