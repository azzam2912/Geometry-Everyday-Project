        \tkzDrawSegment(C,D)
        \tkzDrawSegment(C,S)
        \tkzDrawSegments(E,D D,F)
        \tkzDrawSegments(A,S S,B)
        \tkzDrawSegments[dashed,red](E,S S,F)
        \tkzDrawCircle(O,A)

        \tkzMarkRightAngle[blue, size=0.1364](S,A,C)
