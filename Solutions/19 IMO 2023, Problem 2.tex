\begin{proof}[\textbf{Solusi.}] (Sunday, 23 July 2023)
    Misalkan $I \neq P$ adalah perpotongan $PD$ dengan $\Omega$. Misalkan pula $F$ adalah perpotongan kedua dari garis bagi dalam $\angle BAC$ dengan $\Omega$. Definisikan $H$ sebagai perpotongan $PI$ dengan $AF$. Terakhir, definisikan $Q$ sebagai perpotongan $AF$ dengan $BS$. Akan dibuktikan bahwa garis $PQ$ meyinggung $\omega$ di $P$. 

    \begin{lemma}
        $EI \parallel BC$ 
        \begin{proof}[\textbf{Bukti 1.}] Dengan Rheim Theorem, kita punya $\dangle IEL = \dangle IEB = \dangle IPB = \dangle DPB = \dangle DLB = \dangle DLE$ yang jelas membuktikan bahwa $EI \parallel LD \parallel BC$.
        \end{proof}
        \begin{proof}[\textbf{Bukti 2.}] Karena $BC \parallel LD$ maka $\dangle EBC = \dangle BLD = \dangle BPD = \dangle BPI$ sehingga panjang busur $BEI$ sama dengan panjang busur $EIC$ yang menyebabkan $BE=IC$. Dari sini jelas bahwa $BEIC$ adalah trapesium sama kaki sehingga $EI \parallel BC$.
        \end{proof}
    \end{lemma}

    Karena $AE \perp BC$ maka $\dangle IEA = 90^\circ$ hal ini menunjukkan bahwa $IA$ adalah diameter $\Omega$. Jelas dari fakta tersebut bahwa $\dangle APH = \dangle API = 90^\circ$. Oleh karena itu $\triangle APH$ siku-siku di $P$. 

    Sekarang, observasi bahwa $F$ merupakan titik tengah busur $BC$ yang tidak mengandung $A$. Oleh karena itu, $FS$ merupakan garis sumbu $BC$ dan juga merupakan diameter $\Omega$ sehingga $BC \perp SF \implies SF \parallel AE$. Oleh karena itu, didapat $AS=EF=FI$ yang kembali menyebabkan $AF \parallel SI$. Lebih jauh lagi, hal ini menyebabkan pula $AS \perp AF$ dan $DI \perp IF$ yang berarti $ASIF$ adalah sebuah persegi panjang.

    \begin{lemma}
        $AQ=QH$
        \begin{proof}[\textbf{Bukti 1.}] Misalkan $IS$ berpotongan dengan $AE$ di $X$. Definisikan $M = \operatorname{mid}(BC)$ yang juga dilalui $SF$. Karena $EX \parallel MS$ maka $\dfrac{IX}{IS} = \dfrac{IE}{IM} = 2$. Selanjutnya, karena $AF \parallel SI \implies AH \parallel IX$ maka dari homothety yang berpusat di $D$ dan mengirim $AH$ ke $XI$ maka $\dfrac{AQ}{QH} = \dfrac{XS}{SI}= 1$ atau $AQ=QH$.
        \end{proof}
        \begin{proof}[\textbf{Bukti 2.}] Misalkan $Y$ adalah perpotongan $SA$ dengan $IP$. Karena $AI$ adalah diameter $\Omega$ dan $AE$ adalah garis tinggi $\triangle ABC$, maka $AF$ juga merupakan garis bagi $\angle EAI$. Di lain sisi, karena $\dangle YAF = \dangle SAF = 90^\circ$ dan $AF$ adalah garis bagi dalam $\angle EAI$, maka $AY$ adalah garis bagi luar $\angle EAI$. Oleh karena itu, dengan teorema garis bagi, kita punya $(Y,H;D,I)=-1$. Di sisi lain karena $SI \parallel AF$, jika $P_\infty$ adalah \textit{point at infinity}, maka
        \begin{align*}
             (A,H,Q,P_\infty)\stackrel{S}{=} (Y,H;D,I) = -1  \implies AQ=QH
        \end{align*}
        \end{proof}
    \end{lemma}

    Sadari bahwa $PDHB$ siklis karena $\angle PBQ = \angle PBS = \angle PIS = \angle PHQ$. Selanjutnya, karena $AQ=QH$ dan $\triangle APH$ siku-siku di $P$, maka $\dangle DPQ = \dangle HPQ = \dangle QHP = \dangle QBP = \dangle QBP$, sehingga dari \textit{alternate segment theorem} didapatkan bahwa $QP$ menyinggung $\omega$ di $Q$. Terbukti.
\end{proof}
